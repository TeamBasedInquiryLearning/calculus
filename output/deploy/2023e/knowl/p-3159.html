\begin{equation*} \begin{array}{c|c|c|c|c|c|c|c} f(0) & f'(0) & f''(0) & f^{(3)}(0) & f^{(4)}(0) & f^{(5)}(0) & f^{(6)}(0) & f^{(7)}(0) \\ \hline \sin(0) & \cos(0) & -\sin(0) & -\cos(0) & \sin(0) & \cos(0) & -\sin(0) & -\cos(0) \\ \hline 0 & 1 & 0 & -1 & 0 & 1 & 0 & -1 \end{array} \end{equation*}
