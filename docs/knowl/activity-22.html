(a)
\begin{equation*} \lim_{x\to-6 } \frac{ x^{2} - 6 \, x + 5 }{ x^{2} - 3 \, x - 18 } \end{equation*}

\begin{equation*} \lim_{x\to-1 } \frac{ x^{2} - 1 }{ x^{2} + 3 \, x + 2 } \end{equation*}
(c)
\begin{equation*} \lim_{x\to5 } \frac{ x - 5 }{ \sqrt{x + 31} - 6 } \end{equation*}
