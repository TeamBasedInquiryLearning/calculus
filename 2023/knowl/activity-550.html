(a)
\(\displaystyle \sum_{n=1}^\infty \frac{4 \, \left(-1\right)^{n + 1} n^{2}}{2 \, n^{3} + 4 \, n^{2} + 5}.\)
(b)
\(\displaystyle \sum_{n=1}^\infty \frac{n!}{3 \cdot 3^{n} n^{4}}.\)
